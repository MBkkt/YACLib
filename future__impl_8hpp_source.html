144  util::Result<void> Wrapper(util::Result<T>&& r) noexcept {
145  try {

147  WrapperSubscribe(std::move(r));
148  } else {
149  WrapperOther(std::move(r));
150  }

158  template <typename Arg>
159  void WrapperSubscribe(Arg&& a) {
160  auto future = std::forward<FunctorInvokeT>(_f)(std::forward<Arg>(a));
161  future.GetCore()->SetCallbackInline(_promise.GetCore());
162  std::move(future).Detach();
163  }

165  void WrapperOther(util::Result<T>&& r) {
166  const auto state = r.State();
167  if constexpr (util::IsInvocableV<FunctorInvokeT, T>) {
168  if (state == util::ResultState::Value) {
169  return WrapperSubscribe(std::move(r).Value());
170  }
171  if (state == util::ResultState::Error) {
172  return std::move(_promise).Set(std::move(r).Error());
173  }
174  assert(state == util::ResultState::Exception);
175  return std::move(_promise).Set(std::move(r).Exception());
176  /** We can't use this strategy for other FunctorInvokeT,
177  * because in that case user will not have compiled error for that case:
178  * yaclib::MakeFuture(32) // state == util::ResultState::Value
179  * .ThenInline([](std::exception_ptr/std::error_code) -> yaclib::Future<double> {
180  * throw std::runtime_error{""};
181  * })
182  * .ThenInline([](yaclib::util::Result<double>) { // need double Value
183  * return 1;
184  * });
185  */
186  } else if constexpr (util::IsInvocableV<FunctorInvokeT, std::error_code>) {
187  if (state == util::ResultState::Error) {
188  return WrapperSubscribe(std::move(r).Error());
189  }
190  return std::move(_promise).Set(std::move(r));
191  } else if constexpr (util::IsInvocableV<FunctorInvokeT, std::exception_ptr>) {
192  if (state == util::ResultState::Exception) {
193  return WrapperSubscribe(std::move(r).Exception());
194  }
195  return std::move(_promise).Set(std::move(r));
196  }
197  }
